# ----------- Global Definitions -----------
SET(LIBRARY_NAME luvk)

CMAKE_MINIMUM_REQUIRED(VERSION 3.28)
PROJECT(${LIBRARY_NAME} VERSION 0.0.1 LANGUAGES CXX)

IF (POLICY CMP0048)
    CMAKE_POLICY(SET CMP0048 NEW)
ENDIF ()

# ------------- Library Setup --------------
SET(LUVK_SOURCE_BASE_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/src)
SET(LUVK_INCLUDE_BASE_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/include/)

SET(SOURCE_FILES
		"${LUVK_SOURCE_BASE_DIRECTORY}/Core/Renderer.cpp"
		"${LUVK_SOURCE_BASE_DIRECTORY}/Core/Device.cpp"
		"${LUVK_SOURCE_BASE_DIRECTORY}/Core/RenderGraph.cpp"
		"${LUVK_SOURCE_BASE_DIRECTORY}/Core/Extensions.cpp"

		"${LUVK_SOURCE_BASE_DIRECTORY}/Types/Layer.cpp"
		"${LUVK_SOURCE_BASE_DIRECTORY}/Types/Pair.cpp"
		"${LUVK_SOURCE_BASE_DIRECTORY}/Types/Array.cpp"

		"${LUVK_SOURCE_BASE_DIRECTORY}/Libraries/Compile.cpp"
		"${LUVK_SOURCE_BASE_DIRECTORY}/Libraries/Functional.cpp"

		"${LUVK_SOURCE_BASE_DIRECTORY}/Module.cpp"
)

SET(INCLUDE_FILES
		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Core/Renderer.hpp"
		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Core/Device.hpp"
		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Core/RenderGraph.hpp"
		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Core/Extensions.hpp"

		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Types/Layer.hpp"
		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Types/Pair.hpp"
		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Types/Array.hpp"

		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Libraries/Compile.hpp"
		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Libraries/Functional.hpp"

		"${LUVK_INCLUDE_BASE_DIRECTORY}/luvk/Module.hpp"
)

ADD_LIBRARY(${LIBRARY_NAME} SHARED ${SOURCE_FILES})
SET_TARGET_PROPERTIES(${LIBRARY_NAME} PROPERTIES LINKER_LANGUAGE CXX)

TARGET_INCLUDE_DIRECTORIES(${LIBRARY_NAME} PRIVATE ${LUVK_SOURCE_BASE_DIRECTORY})
TARGET_INCLUDE_DIRECTORIES(${LIBRARY_NAME} PUBLIC ${LUVK_INCLUDE_BASE_DIRECTORY})

TARGET_COMPILE_DEFINITIONS(${LIBRARY_NAME} PRIVATE BUILD_DLL)

IF (WIN32)
	SET(VOLK_STATIC_DEFINES VK_USE_PLATFORM_WIN32_KHR)
ELSEIF (MACOS)
	SET(VOLK_STATIC_DEFINES VK_USE_PLATFORM_MACOS_MVK)
ELSEIF (UNIX)
	SET(VOLK_STATIC_DEFINES VK_USE_PLATFORM_XCB_KHR)
ENDIF (WIN32)

ADD_DEFINITIONS(-DVK_NO_PROTOTYPES=1)
ADD_DEFINITIONS(-DRPS_VK_RUNTIME=1)
ADD_DEFINITIONS(-DRPS_VK_DYNAMIC_LOADING=1)
ADD_DEFINITIONS(-DVMA_STATIC_VULKAN_FUNCTIONS=0)
ADD_DEFINITIONS(-DVMA_DYNAMIC_VULKAN_FUNCTIONS=1)

INCLUDE(FetchContent)

FETCHCONTENT_DECLARE(
		volk
		GIT_REPOSITORY https://github.com/zeux/volk.git
		GIT_TAG master
)

FETCHCONTENT_DECLARE(
		rps
		GIT_REPOSITORY https://github.com/GPUOpen-LibrariesAndSDKs/RenderPipelineShaders.git
		GIT_TAG main
)

FETCHCONTENT_MAKEAVAILABLE(
		volk
		rps
)

TARGET_LINK_LIBRARIES(${LIBRARY_NAME} PUBLIC
		volk
		# rps_core		- Not using yet
		# rps_frontend  - Not using yet
		# rps_runtime   - Not using yet
		rps_runtime_vkdyn
)

TARGET_INCLUDE_DIRECTORIES(${LIBRARY_NAME} PUBLIC
		${volk_SOURCE_DIR}
		${rps_SOURCE_DIR}/include
)
